Basic Query

query {
  users {
    data {
      id
      name
      email
      created_at
    }
    paginatorInfo {
      count
      total
    }
  }
}

Search by Email

query {
  users(email: "%gmail.com%") {
    data {
      id
      name
      email
      created_at
    }
    paginatorInfo {
      count
      total
    }
  }
}
